   ```typescript
   import { z } from 'zod';

   export const createRoomSchema = z.object({
     name: z.string().min(1).max(60),
     topic: z.string().max(280).optional(),
   });

   export const createAgentSchema = z.object({
     name: z.string().min(1).max(60),
     avatarColor: z.string().regex(/^#[0-9A-Fa-f]{6}$/),
     avatarIcon: z.string().min(1),
     promptRole: z.string().min(1),
     promptPersonality: z.string().optional(),
     promptRules: z.string().optional(),
     promptConstraints: z.string().optional(),
     provider: z.enum(['anthropic', 'openai', 'google', 'openrouter', 'ollama']),
     model: z.string().min(1),
     temperature: z.number().min(0).max(1).default(0.7),
     presetId: z.string().optional(),
   });

   export const updateAgentSchema = createAgentSchema.partial();

   export const addAgentToRoomSchema = z.object({
     agentId: z.string().min(1),
   });

   export const removeAgentFromRoomSchema = z.object({
     roomAgentId: z.string().min(1),
   });

   export const saveProviderKeySchema = z.object({
     apiKey: z.string().optional(),
     baseUrl: z.string().url().optional(),
   });
   ```

   - **POST**: Parse body with `createRoomSchema.safeParse()`. If invalid, return 400 with `{ error: issues }`. Generate id with `nanoid()`. Insert into rooms table. Return 201 with the created room object.

3. Create `src/app/api/rooms/[roomId]/route.ts`:

   - **DELETE**: Delete room by id: `db.delete(rooms).where(eq(rooms.id, roomId))`. CASCADE handles room_agents and messages. Return `{ ok: true }`.

4. Create `src/app/api/rooms/[roomId]/agents/route.ts`:
   - **POST**: Parse body with `addAgentToRoomSchema`. Fetch library agent by agentId from body. If not found, return 404. Copy ALL config columns from library agent into new roomAgents row (EXACT pattern from RESEARCH.md "Copy Agent to Room"):
     - Copy: name, avatarColor, avatarIcon, promptRole, promptPersonality, promptRules, promptConstraints, provider, model, temperature
     - Set: id=nanoid(), roomId from URL param, sourceAgentId=source.id
     - Calculate position: count existing roomAgents for this room, set position = count
   - Return 201 with the created roomAgent object.
   - **DELETE**: Parse body with `removeAgentFromRoomSchema`. Delete the roomAgent row by roomAgentId. Return `{ ok: true }`.

5. Create `src/app/api/agents/route.ts`:
   - `export const dynamic = 'force-dynamic';`
   - **GET**: Query all agents ordered by createdAt desc. Return JSON array.
   - **POST**: Parse body with `createAgentSchema`. Generate id with `nanoid()`. Insert into agents table. Return 201 with created agent.

6. Create `src/app/api/agents/[agentId]/route.ts`:
   - `export const dynamic = 'force-dynamic';`
   - **GET**: Find agent by id. If not found, return 404. Return agent object.
   - **PUT**: Parse body with `updateAgentSchema`. Update agent fields using `db.update(agents).set({ ...parsed, updatedAt: new Date() }).where(eq(agents.id, agentId))`. Return updated agent.
   - **DELETE**: Delete agent by id. Return `{ ok: true }`.

All route handlers must:
