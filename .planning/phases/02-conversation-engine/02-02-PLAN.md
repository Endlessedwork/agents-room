Create `src/lib/conversation/manager.ts`:

```typescript
import { db as defaultDb } from '@/db';
import { rooms, roomAgents, messages } from '@/db/schema';
import { eq, desc, and, count } from 'drizzle-orm';
import { nanoid } from 'nanoid';
import { streamLLM } from '@/lib/llm/gateway';
import { ContextService } from './context-service';
import { SpeakerSelector, type RoomAgentRow } from './speaker-selector';

// In-memory registry of active AbortControllers (per room)
const activeControllers = new Map<string, AbortController>();

export class ConversationManager {
  // Accept db parameter for testability (defaults to production singleton)
  static async start(roomId: string, db = defaultDb): Promise<void> { ... }
  static async pause(roomId: string, db = defaultDb): Promise<void> { ... }
  static async stop(roomId: string, db = defaultDb): Promise<void> { ... }
  static async resume(roomId: string, db = defaultDb): Promise<void> { ... }
  // Exposed for testing
  static _getActiveController(roomId: string): AbortController | undefined { ... }
  static _clearControllers(): void { ... }
}
```

**start(roomId, db):**
1. If `activeControllers.has(roomId)`, return early (prevent double-start)
2. Load room via `db.query.rooms.findFirst({ where: eq(rooms.id, roomId) })`
3. Throw if room not found
4. Set `status = 'running'` via `db.update(rooms).set({ status: 'running' })`
5. Load room agents ordered by position: `db.query.roomAgents.findMany({ where: eq(roomAgents.roomId, roomId), orderBy: ... })`
6. Throw if no agents
7. Create `SpeakerSelector(agents, room.speakerStrategy, getProviderConfig)` where `getProviderConfig` queries `providerKeys` table
8. Fire-and-forget IIFE with `.catch(err => console.error('[ConversationManager]', err))`
9. Inside loop: `while (turnCount < turnLimit)`
   a. Re-query room status; break if not 'running'
   b. Get next agent from selector
   c. Build context via `ContextService.buildContext(db, roomId, agent)`
   d. Create `AbortController`, store in `activeControllers`
   e. Call `streamLLM({ provider, model, config, system, messages, temperature, abortSignal })`
   f. Consume `result.textStream` to build `fullText`
   g. Await `result.usage` for token counts
   h. In catch: if `err.name === 'AbortError'` break; else persist system error message and break
   i. In finally: `activeControllers.delete(roomId)`
   j. Persist message: `db.insert(messages).values({ id: nanoid(), roomId, roomAgentId: agent.id, role: 'agent', content: fullText, model: agent.model, inputTokens: usage.inputTokens ?? null, outputTokens: usage.outputTokens ?? null })`
   k. Update `lastActivityAt`
   l. Check `ContextService.detectRepetition(db, roomId)`; if true, set status='paused', persist warning, break
   m. Increment turnCount
10. After loop: if room still 'running', set to 'idle'

**pause(roomId, db):** Set `status = 'paused'`. Loop exits on next iteration status check.

**stop(roomId, db):** Set `status = 'idle'`. Get controller from map, call `controller.abort()`, delete from map.

**resume(roomId, db):**
1. Load room
2. Count existing agent messages: `SELECT COUNT(*) FROM messages WHERE roomId = ? AND role = 'agent'`
3. Calculate `remainingTurns = room.turnLimit - existingCount`
4. If remainingTurns <= 0, return (conversation already at limit)
5. Call `start()` internally but with `turnLimit` overridden to `remainingTurns`

}

async function waitForMessages(db, roomId, target, timeoutMs = 5000) {
  const start = Date.now();
  while (Date.now() - start < timeoutMs) {
    const [{ value }] = await db.select({ value: count(messages.id) }).from(messages).where(eq(messages.roomId, roomId));
    if (value >= target) return;
    await new Promise(r => setTimeout(r, 50));
  }
  throw new Error(`Timed out waiting for ${target} messages`);
}
